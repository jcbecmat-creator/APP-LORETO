Al forzar el windowWidth en html2canvas, // obligamos a que todo este bloque se escale hacia abajo para caber en la hoja Carta. let htmlContent = `

Loreto

Ingeniería, Construcción, Edificación y Remodelación

REPORTE DE VACACIONES DEL PERSONAL

`; dataPDF.forEach(row => { const gen = row.dias_generados || 0; htmlContent += ` `; let usados = 0; for(let i=1; i<=4; i++) { const p = row[`periodo_${i}`]; if(p && p.inicio) { let est = p.estado === 'aprobado' ? 'Aprobado' : (p.estado === 'modificado' ? 'Aprobado(M)' : (p.estado === 'rechazado' ? 'Rechazado' : 'Pendiente')); htmlContent += ` `; usados += parseInt(p.dias); } else { htmlContent += ``; } } const saldo = gen - usados; htmlContent += ` `; }); htmlContent += `
TRABAJADOR DÍAS
AUT.		 1ER PERIODO 2DO PERIODO 3ER PERIODO 4TO PERIODO USADOS SALDO
FechasDíasEstatus FechasDíasEstatus FechasDíasEstatus FechasDíasEstatus
${row.nombre || row.id} ${gen}${formateaFechaCorta(p.inicio)} - ${formateaFechaCorta(p.fin)} ${p.dias} ${est}---${usados} ${saldo}
Reporte generado el: ${new Date().toLocaleString('es-MX')}
`; const opt = { margin: [10, 10, 10, 10], filename: `Reporte_Vacaciones_${Date.now()}.pdf`, image: { type: 'jpeg', quality: 0.98 }, // LA MAGIA OCURRE AQUÍ: Forzamos el "viewport" del fotógrafo a 1200px. // Al ser más grande que la tabla (1150px), garantiza que TODO entra en la foto antes de reducirse a la hoja. html2canvas: { scale: 2, useCORS: true, windowWidth: 1200 }, jsPDF: { unit: 'mm', format: 'letter', orientation: 'landscape' } }; const btn = document.getElementById('btn-descargar-pdf'); const oldTxt = btn.innerHTML; btn.innerHTML = ' Generando...'; lucide.createIcons(); html2pdf().set(opt).from(htmlContent).save().then(() => { btn.innerHTML = oldTxt; lucide.createIcons(); }); }
